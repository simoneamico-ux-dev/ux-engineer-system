<TabItem value="js" label="script.js">

```js
const currentDateParagraph = document.getElementById("current-date");
const dateOptionsSelectElement = document.getElementById("date-options");

const date = new Date();
const day = date.getDate();
const month = date.getMonth() + 1;
const year = date.getFullYear();
const hours = date.getHours();
const minutes = date.getMinutes();

const formattedDate = `${day}-${month}-${year}`;
currentDateParagraph.textContent = formattedDate;

dateOptionsSelectElement.addEventListener("change", () => {

switch (dateOptionsSelectElement.value) {
    case "yyyy-mm-dd":
    currentDateParagraph.textContent = formattedDate
        .split("-")
        .reverse()
        .join("-");
    break;
    case "mm-dd-yyyy-h-mm":
    currentDateParagraph.textContent = `${month}-${day}-${year} ${hours} Hours ${minutes} Minutes`;
    break;
    default:
    currentDateParagraph.textContent = formattedDate;
}
});
```

      // Result: "25-12-2024" for example

      // ğŸ“ SHOW THE DATE: Inserts into paragraph

        // ğŸ§ LISTEN FOR CHANGES: When user selects an option
        // "change" = event triggered when value changes
        // () => {} = arrow function, modern syntax
        
        // ğŸ”€ ===== SWITCH FOR DIFFERENT FORMATS =====
        switch (dateOptionsSelectElement.value) {
          // ğŸ”€ SWITCH: Checks the selected value
          // .value = the value of the selected option
          
          case "yyyy-mm-dd":
            // ğŸ“… CASE 1: Year-month-day format
            
            currentDateParagraph.textContent = formattedDate
              .split("-")
              .reverse()
              .join("-");
            // ğŸ”„ STRING MANIPULATION:
            // .split("-") = splits "25-12-2024" into ["25", "12", "2024"]
            // .reverse() = reverses the array ["2024", "12", "25"]
            // .join("-") = joins with "-" = "2024-12-25"
            // It's like taking apart and reassembling the pieces!
            
            break;
            // ğŸ›‘ BREAK: Exits the switch
            
          case "mm-dd-yyyy-h-mm":
            // ğŸ“… CASE 2: Format with time
            
            currentDateParagraph.textContent = `${month}-${day}-${year} ${hours} Hours ${minutes} Minutes`;
            // ğŸ“ COMPLEX FORMAT: Includes time
            // Template literal to build the string
            // Result: "12-25-2024 14 Hours 30 Minutes"
            
            break;
            // ğŸ›‘ BREAK: Exits the switch
            
          default:
            // ğŸ“… DEFAULT: Any other case (including "dd-mm-yyyy")
            
            currentDateParagraph.textContent = formattedDate;
            // ğŸ“ ORIGINAL FORMAT: Day-month-year
            // Reuses the already calculated variable
        }
        // End of switch
      });
      // End of event listener
