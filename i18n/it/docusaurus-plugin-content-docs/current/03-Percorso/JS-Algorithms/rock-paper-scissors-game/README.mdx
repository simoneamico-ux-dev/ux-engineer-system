```js
function getRandomComputerResult() {
const options = ["Rock", "Paper", "Scissors"];
const randomIndex = Math.floor(Math.random() * options.length);
return options[randomIndex];
}

function hasPlayerWonTheRound(player, computer) {
return (
    (player === "Rock" && computer === "Scissors") ||
    (player === "Scissors" && computer === "Paper") ||
    (player === "Paper" && computer === "Rock")
);
}

let playerScore = 0;
let computerScore = 0;

function getRoundResults(userOption) {
const computerResult = getRandomComputerResult();

if (hasPlayerWonTheRound(userOption, computerResult)) {
    playerScore++;
    return `Player wins! ${userOption} beats ${computerResult}`;
} else if (computerResult === userOption) {
    return `It's a tie! Both chose ${userOption}`;
} else {
    computerScore++;
    return `Computer wins! ${computerResult} beats ${userOption}`;
}
}

const playerScoreSpanElement = document.getElementById("player-score");
const computerScoreSpanElement = document.getElementById("computer-score");
const roundResultsMsg = document.getElementById("results-msg");
const winnerMsgElement = document.getElementById("winner-msg");
const optionsContainer = document.querySelector(".options-container");
const resetGameBtn = document.getElementById("reset-game-btn");

function showResults(userOption) {
roundResultsMsg.innerText = getRoundResults(userOption);
computerScoreSpanElement.innerText = computerScore;
playerScoreSpanElement.innerText = playerScore;

if (playerScore === 3 || computerScore === 3) {
    winnerMsgElement.innerText = `${
    playerScore === 3 ? "Player" : "Computer"
    } has won the game!`;

    resetGameBtn.style.display = "block";
    optionsContainer.style.display = "none";
}

};
function resetGame() {
playerScore = 0;
computerScore = 0;
playerScoreSpanElement.innerText = "0";
computerScoreSpanElement.innerText = "0";
resetGameBtn.style.display = "none";
optionsContainer.style.display = "block";
winnerMsgElement.innerText = "";
roundResultsMsg.innerText = "";
};

resetGameBtn.addEventListener("click", resetGame);

const rockBtn = document.getElementById("rock-btn");
const paperBtn = document.getElementById("paper-btn");
const scissorsBtn = document.getElementById("scissors-btn");

rockBtn.addEventListener("click", function () {
showResults("Rock");
});

paperBtn.addEventListener("click", function () {
showResults("Paper");
});

scissorsBtn.addEventListener("click", function () {
showResults("Scissors");
});
```

      /* üé≤ ===== FUNZIONE: Scelta Random del Computer ===== */
      function getRandomComputerResult() {
        const options = ["Rock", "Paper", "Scissors"];
        // üìã ARRAY OPZIONI: Le tre scelte possibili
        
        const randomIndex = Math.floor(Math.random() * options.length);
        // üé≤ NUMERO RANDOM: La formula magica!
        // Math.random() = numero tra 0 e 0.999...
        // * options.length = moltiplica per 3 (numero di opzioni)
        // Math.floor() = arrotonda per difetto
        // Risultato: 0, 1, o 2 (indici dell'array)
        
        return options[randomIndex];
        // üéØ RITORNA: L'opzione scelta casualmente
      }
      
      /* üèÜ ===== FUNZIONE: Chi Vince il Round? ===== */
      function hasPlayerWonTheRound(player, computer) {
        return (
          (player === "Rock" && computer === "Scissors") ||
          (player === "Scissors" && computer === "Paper") ||
          (player === "Paper" && computer === "Rock")
        );
        // üßÆ LOGICA VITTORIA: Tutte le combinazioni vincenti
        // || = OR logico, basta una condizione vera
        // && = AND logico, entrambe devono essere vere
        // Ritorna true se il giocatore vince, false altrimenti
      }
      
      /* üìä ===== VARIABILI GLOBALI: I Contatori ===== */
      let playerScore = 0;
      let computerScore = 0;
      // üî¢ PUNTEGGI: Iniziano da zero, let perch√© cambieranno
      
      /* üéÆ ===== FUNZIONE: Gestione del Round ===== */
      function getRoundResults(userOption) {
        const computerResult = getRandomComputerResult();
        // ü§ñ SCELTA COMPUTER: Chiama la funzione random
      
        if (hasPlayerWonTheRound(userOption, computerResult)) {
          playerScore++;
          // üë§ VITTORIA GIOCATORE: Incrementa il punteggio
          // ++ = incremento di 1
          
          return `Player wins! ${userOption} beats ${computerResult}`;
          // üì¢ MESSAGGIO VITTORIA: Template literal con variabili
        
        } else if (computerResult === userOption) {
          return `It's a tie! Both chose ${userOption}`;
          // ü§ù PAREGGIO: Stessa scelta
        
        } else {
          computerScore++;
          // ü§ñ VITTORIA COMPUTER: Incrementa il suo punteggio
          
          return `Computer wins! ${computerResult} beats ${userOption}`;
          // üì¢ MESSAGGIO SCONFITTA: Il computer ha vinto
        }
      }

      /* üé≠ ===== FUNZIONE: Mostra i Risultati ===== */
      function showResults(userOption) {

        // üìù MOSTRA RISULTATO: Aggiorna il messaggio del round
